@macro table4{title1,title2,title3,title4}
@set table-title2 \title2\
@set table-title3 \title3\
@set table-title4 \title4\
@end macro

@end macro

@end ifset
@ifset table-title4
@*- @value{table-title4}: \col4\
@end ifset
@sp 1
@end macro
